<script>
var url = '/zettelkasten/graph.json';
var Graph = ForceGraph()(document.getElementById('graph'))
  .cooldownTicks(100);

fetch(url)
.then(res => res.json())
.then(out => {
Graph.graphData(out)
     .d3Force('center', null)
     .onEngineStop(() => Graph.zoomToFit(400));
}
);
</script>
